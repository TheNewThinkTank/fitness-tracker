    """_summary_

    :param date: _description_, defaults to "2022-06-27"
    :type date: str, optional
    :return: _description_
    :rtype: tuple[str, str]
    """

    month_zeropadded = date[5:7]
    month = int(month_zeropadded.removeprefix("0"))
    MONTH = Months(month).name.capitalize()
    YEAR = date[:4]

    return YEAR, MONTH
